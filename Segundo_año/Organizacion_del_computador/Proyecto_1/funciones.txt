Cuadrado:

	mov x13, 640
	sub x13, x13, x4				// x13 = 640 - x4
	lsl x13, x13, 2					// x13 = (640 - x4) * 4

	mov x2, x5

	mov x1, x4 

Cuadrado2:

	stur w10,[x0]  					// Colorear el pixel N
	add x0, x0, 4					// Siguiente pixel
	sub x1, x1, 1    				// Decrementar contador X
	cbnz x1, Cuadrado2
	add x0,x0, x13		  			// Si no terminó la fila, salto
	sub x2, x2, 1    				// Decrementar contador Y
	cbnz x2, Cuadrado1  			// Si no es la última fila, salto

	mov x0,x20

//------------------------------------------ Circulo ------------------------------------------//


	Circulo:										// x4 = Tamaño x     x5 = Tamaño y		

		mov x16, xzr
		mov x17, xzr			
		mov x18, xzr
		mov x21, xzr
		mov x22, xzr					
		mov x24, x14								// x24 = x0
		mov x25, x15								// x25 = y0
		mov x26, xzr				 				// x26 = x1
		mov x27, xzr								// x27 = y1
	
		sub x26, x14, x9							// En x26 guardo las cordenadas de x del inicio del cuadrado  "Le resto el radio"
		sub x27, x15, x9							// En x27 guardo las cordenadas de y del inicio del cuadrado  "Le resto el radio"
		sub sp,sp, #8  
		str x30,[sp,#0]

		// "pinto el cuadrado dependiendo si cumple las condiciones" de aca en adelante es muy parecido a la funcion cuadrado solo que pregunto si cumple la condicion de calculo para pintar el pixel

		lsl x21,x9,1								//Ysize

		loop_circ1:

			mov x14, x26
			mov x15, x27

			bl Calcular_Direccion

			lsl x22,x9,1							//Xsize

		loop_circ2:

			bl Calculo								//hago los calculos necesarios con pitagora para ver si la hipotenusa geenerada por la linea entre el centro del circulo y el pixel que estoy viendo 
			b.le Colorear							//coloreo el circulo
			bl Calcular_Direccion					//calculo la direccion de mi x0

		Siguiente:	

			add x0, x0, 4  							// Siguiente pixel
			add x26, x26, 1							// Siguiente coordenada en x
			sub x22, x22, 1  	  					// Decrementar contador de mi diametro
		
			cbnz x22,loop_circ2 					// Si no terminó la fila, salto
		
			sub x21, x21, 1   						// Decrementar contador Y
			add x27, x27, 1							// Siguiente pixel en Y
			sub x26, x24, x9 						// Pongo devuelta mi coordenadad de x En 0
			msub x0, x3, x9, x0						// Vuelvo al princio de la linea que estoy pintado x0=x0-x1*4
			madd x0, x1, x3, x0  					// x0 = x0 + 640 * 4

  			cbnz x21,loop_circ1  					// Si no es la última fila, saltostur w11,[x20]

			ldr x30,[sp,#0]  						// Cargo x15 y x14 a sus valores originales y decremento la pila en 2  
			add sp,sp, #8

			mov x0,x20

		ret

		Colorear:

			stur w10,[x0]  							// Colorear el pixel N
			b Siguiente								// Sigo viendo mis demas pixeles

		Calculo:		
			
			mov x16,x26								// Guardo mis x1 y1
			mov x17,x27
				
			sub x16, x16, x24						// x25 = x1 - x0
			mul x16, x16, x16						// x25 = x25 al cuadrado
			sub x17, x17, x25						// x26 = y1 - y0
			mul x17, x17, x17						// x26 = x26 al cuadrado

			add x16, x17, x16						// x25 = x26 + x25
			mov x17, x9								// x26 ahora es el radio

			mul x17, x17, x17						// x26 es radio cuadrado
			cmp x16, x17							// x25 = x25 - diametro

			b.gt Siguiente							// Si mi radio cuadrado en mayor extricto que mi x25 NO pinto ese pixel
			b Colorear								// Si es menor lo pinto
